title template_and_instance_creation (TXT file can be changed on sequencediagram.org)


actor Manager

participant ClassroomTemplate
participant ProjectTemplate
participant TaskTemplate

actor Instructor

participant ClassroomInstance
participant ProjectInstance
participant TaskInstance

loop Mangager wants to create more ClassroomTemplates
Manager -->>*ClassroomTemplate: <<create>>
Manager->ClassroomTemplate:add project instance to classroom instance

loop Mangager wants to create more ProjectTemplates
ClassroomTemplate-->>*ProjectTemplate:<<create>>
ClassroomTemplate->ClassroomTemplate:add ProjectTemplate

loop Manager wants to create more Tasks
Manager->ClassroomTemplate:add task instance to project instance
ClassroomTemplate->ProjectTemplate:add task instance to project instance
ProjectTemplate-->>*TaskTemplate:<<create>>
ProjectTemplate->ProjectTemplate:add TaskInstance
end
end
end

loop Instructor wants to create more Classrooms
Instructor-->>*ClassroomInstance:<<create>> from template
loop ClassroomTemplate contains uncreated ProjectTemplates
ClassroomInstance-->>*ProjectInstance:<<create>> from template 
loop ProjectTemplate contains uncreated TaskTemplates
ProjectInstance-->>*TaskInstance:<<create>> from template
end
end
end
